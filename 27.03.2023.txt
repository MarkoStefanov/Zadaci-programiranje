#include <iostream>
#include <deque>

using namespace std;

int main()
{
    int n, d, x;
    cin>>n>>d;
    deque<int> dq;

    for(int i=0; i<n; i++)
    {
        cin>>x;
        dq.push_back(x);
    }
    for(int i=0; i<d; i++)
    {
        dq.push_front(dq.back());
        dq.pop_back();
    }
    auto it=dq.begin();
    while(it!=dq.end())
    {
        cout<<*it<<' ';
        it++;
    }
    return 0;
}

#include <iostream>
#include <list>
using namespace std;

int main()
{
    list<int> one= {1, 2, 5, 6}, two= {1, 3, 8}, result;
    for(int i=0; i<7; i++)
    {
        if(!one.empty()&&one.front()<two.front())
        {
            result.push_back(one.front());
            one.pop_front();
        }
        else
        {
            result.push_back(two.front());
            two.pop_front();
        }
    }
    auto it=result.begin();
    while(it!=result.end())
    {
        cout<<*it<<' ';
        it++;
    }
    return 0;
}

#include <iostream>
#include <stack>

using namespace std;

int main()
{
    string s;
    cin>>s;
    stack<char> zagradi;
    for(int i=0; i<s.size(); i++)
    {
        if(s[i]=='('||s[i]=='{'||s[i]=='[')
            zagradi.push(s[i]);
        else if(s[i]==']')
        {
            if(!zagradi.empty()&&zagradi.top()=='[')
                zagradi.pop();
            else
            {
                zagradi.push(s[i]);
                break;
            }
        }
        else if(s[i]=='}')
        {
            if(!zagradi.empty()&&zagradi.top()=='{')
                zagradi.pop();
            else
            {
                zagradi.push(s[i]);
                break;
            }
        }
        else if(s[i]==')')
        {
            if(!zagradi.empty()&&zagradi.top()=='(')
                zagradi.pop();
            else
            {
                zagradi.push(s[i]);
                break;
            }
        }
    }
    if(zagradi.empty())
        cout<<"Korektno";
    else
        cout<<"Nekorektno";

    return 0;
}
